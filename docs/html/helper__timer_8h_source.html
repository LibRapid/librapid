83class StopWatchWin : public StopWatchInterface {
84public:
86 StopWatchWin() :
87 start_time(), end_time(), diff_time(0.0f), total_time(0.0f), running(false),
88 clock_sessions(0), freq(0), freq_set(false) {
89 if (!freq_set) {
90 // helper variable
91 LARGE_INTEGER temp;
92
93 // get the tick frequency from the OS
94 QueryPerformanceFrequency(reinterpret_cast<LARGE_INTEGER *>(&temp));
95
96 // convert to type in which it is needed
97 freq = (static_cast<double>(temp.QuadPart)) / 1000.0;
98
99 // rememeber query
100 freq_set = true;
101 }
102 }
103
104 // Destructor
105 ~StopWatchWin() {}
106
107public:
109 inline void start();
110
112 inline void stop();
113
115 inline void reset();
116
120 inline float getTime();
121
124 inline float getAverageTime();
125
126private:
127 // member variables
128
130 LARGE_INTEGER start_time;
132 LARGE_INTEGER end_time;
133
135 float diff_time;
136
138 float total_time;
139
141 bool running;
142
145 int clock_sessions;
146
148 double freq;
149
151 bool freq_set;
152};

159inline void StopWatchWin::start() {
160 QueryPerformanceCounter(reinterpret_cast<LARGE_INTEGER *>(&start_time));
161 running = true;
162}
163
168inline void StopWatchWin::stop() {
169 QueryPerformanceCounter(reinterpret_cast<LARGE_INTEGER *>(&end_time));
170 diff_time = static_cast<float>(
171 ((static_cast<double>(end_time.QuadPart) - static_cast<double>(start_time.QuadPart)) / freq));
172
173 total_time += diff_time;
174 clock_sessions++;
175 running = false;
176}
177
182inline void StopWatchWin::reset() {
183 diff_time = 0;
184 total_time = 0;
185 clock_sessions = 0;
186
187 if (running) { QueryPerformanceCounter(reinterpret_cast<LARGE_INTEGER *>(&start_time)); }
188}
189
196inline float StopWatchWin::getTime() {
197 // Return the TOTAL time to date
198 float retval = total_time;
199
200 if (running) {
201 LARGE_INTEGER temp;
202 QueryPerformanceCounter(reinterpret_cast<LARGE_INTEGER *>(&temp));
203 retval += static_cast<float>(
204 ((static_cast<double>(temp.QuadPart) - static_cast<double>(start_time.QuadPart)) / freq));
205 }
206
207 return retval;
208}
209
214inline float StopWatchWin::getAverageTime() {
215 return (clock_sessions > 0) ? (total_time / clock_sessions) : 0.0f;
216}

337inline float StopWatchLinux::getDiffTime() {
338 struct timeval t_time;
339 gettimeofday(&t_time, 0);
340
341 // time difference in milli-seconds
342 return static_cast<float>(1000.0 * (t_time.tv_sec - start_time.tv_sec) +
343 (0.001 * (t_time.tv_usec - start_time.tv_usec)));
344}
