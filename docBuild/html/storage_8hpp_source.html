148 template<typename A>
149 typename std::allocator_traits<A>::pointer
150 safeAllocate(A &alloc, typename std::allocator_traits<A>::size_type size) {
151 using Traits = std::allocator_traits<A>;
152 using Pointer = typename Traits::pointer;
153 using ValueType = typename Traits::value_type;
154 Pointer ptr = alloc.allocate(size);
155
156 // If the type cannot be trivially constructed, we need to
157 // initialize each value
158 if (!typetraits::TriviallyDefaultConstructible<ValueType>::value) {
159 for (Pointer p = ptr; p != ptr + size; ++p) {
160 Traits::construct(alloc, p, ValueType());
161 }
162 }
163
164 return ptr;
165 }
166
167 template<typename A>
168 void safeDeallocate(A &alloc, typename std::allocator_traits<A>::pointer ptr,
169 typename std::allocator_traits<A>::size_type size) {
170 using Traits = std::allocator_traits<A>;
171 using Pointer = typename Traits::pointer;
172 using ValueType = typename Traits::value_type;
173
174 // If the type cannot be trivially destructed, we need to
175 // destroy each value
176 if (!typetraits::TriviallyDefaultConstructible<ValueType>::value) {
177 for (Pointer p = ptr; p != ptr + size; ++p) { Traits::destroy(alloc, p); }
178 }
179 Traits::deallocate(alloc, ptr, size);
180 }

282 template<typename T, typename A>
283 LIBRAPID_ALWAYS_INLINE void Storage<T, A>::resizeImpl(SizeType newSize) {
284 SizeType oldSize = size();
285 Pointer oldBegin = m_begin;
286 if (oldSize != newSize) {
287 // Reallocate
288 m_begin = detail::safeAllocate(m_allocator, newSize);
289 m_end = m_begin + newSize;
290
291 if constexpr (typetraits::TriviallyDefaultConstructible<T>::value) {
292 // Use a slightly faster memcpy if the type is trivially default constructible
293 std::uninitialized_copy(oldBegin, oldBegin + std::min(oldSize, newSize), m_begin);
294 } else {
295 // Otherwise, use the standard copy algorithm
296 std::copy(oldBegin, oldBegin + std::min(oldSize, newSize), m_begin);
297 }
298
299 detail::safeDeallocate(m_allocator, oldBegin, oldSize);
300 }
301 }
